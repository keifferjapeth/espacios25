      });
    }, { threshold: 0.22 });
    animatedNodes.forEach((node) => reveal.observe(node));

    // FAQ accordion animation for seamless open/close
    const faqs = Array.from(document.querySelectorAll('details.faq'));
    faqs.forEach((faq) => {
      const answer = faq.querySelector('.answer');
      const summary = faq.querySelector('summary');

      const setHeight = (open) => {
        const contentHeight = answer.scrollHeight;
        answer.style.maxHeight = open ? `${contentHeight + 20}px` : '0px';
        answer.style.opacity = open ? '1' : '0';
      };

      // Init state
      setHeight(faq.open);

      summary.addEventListener('click', (event) => {
        event.preventDefault();
        const isOpening = !faq.open;

        // Close all others for a smoother experience
        faqs.forEach((item) => {
          if (item !== faq && item.open) {
            item.open = false;
            const otherAnswer = item.querySelector('.answer');
            otherAnswer.style.maxHeight = '0px';
            otherAnswer.style.opacity = '0';
          }
        });

        faq.open = isOpening;
        setHeight(isOpening);
      });

      window.addEventListener('resize', () => {
        if (faq.open) {
          setHeight(true);
        }
      });
    });
  </script>
</body>
