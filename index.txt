  <script>
    const panel = document.getElementById("panel");
    const speedValue = document.getElementById("speedValue");
    const hueValue = document.getElementById("hueValue");
    const particlesValue = document.getElementById("particlesValue");
    const orbits = [...panel.querySelectorAll(".orbit")];
    const halo = panel.querySelector(".halo");
    const trace = panel.querySelector(".trace");
    const controls = {
      pulse: document.getElementById("pulse"),
      shuffle: document.getElementById("shuffle"),
    };

    const state = {
      speed: 1,
      hue: 204,
      particles: [],
    };

    const createParticle = () => {
      const particle = document.createElement("div");
      particle.className = "particle";
      const angle = Math.random() * Math.PI * 2;
      const radius = 42 + Math.random() * 120;
      const x = 50 + Math.cos(angle) * radius;
      const y = 50 + Math.sin(angle) * radius;
      particle.style.left = `${x}%`;
      particle.style.top = `${y}%`;
      particle.style.animationDelay = `${Math.random() * -3}s`;
      panel.appendChild(particle);
      state.particles.push(particle);
    };

    const spawnParticles = (count) => {
      panel.querySelectorAll(".particle").forEach((p) => p.remove());
      state.particles = [];
      for (let i = 0; i < count; i += 1) createParticle();
      particlesValue.textContent = count;
    };

    spawnParticles(12);

    const setHue = (hue) => {
      document.documentElement.style.setProperty("--accent", `hsl(${hue}, 90%, 70%)`);
      halo.style.filter = `hue-rotate(${hue - 190}deg)`;
      orbits.forEach((orbit, idx) => {
        orbit.style.borderColor = `hsla(${hue + idx * 12}, 80%, 70%, 0.22)`;
      });
      panel.querySelectorAll(".particle").forEach((p, idx) => {
        p.style.background = `linear-gradient(145deg, hsl(${hue + idx * 6}, 90%, 70%), hsl(${hue + 30}, 90%, 72%))`;
      });
      hueValue.textContent = `${Math.round(hue)}°`;
    };

    const setSpeed = (multiplier) => {
      panel.style.setProperty("--speed", multiplier);
      [halo, trace, ...orbits].forEach((el) => {
        el.style.animationDuration = `calc(${parseFloat(getComputedStyle(el).animationDuration)} / ${multiplier})`;
      });
      panel.querySelectorAll(".particle").forEach((p) => {
        p.style.animationDuration = `calc(${parseFloat(getComputedStyle(p).animationDuration)} / ${multiplier})`;
      });
      speedValue.textContent = `${multiplier.toFixed(1)}x`;
    };

    const animatePanel = (e) => {
      const rect = panel.getBoundingClientRect();
      const x = (e.clientX - rect.left) / rect.width - 0.5;
      const y = (e.clientY - rect.top) / rect.height - 0.5;
      const intensity = Math.min(1, Math.hypot(x, y) * 2);
      panel.style.transform = `perspective(900px) rotateX(${y * -6}deg) rotateY(${x * 8}deg) translateZ(0)`;
      halo.style.opacity = 0.7 + intensity * 0.3;
      halo.style.transform = `scale(${1 + intensity * 0.08})`;
    };

    panel.addEventListener("pointermove", animatePanel);
    panel.addEventListener("pointerleave", () => {
      panel.style.transform = "perspective(900px) rotateX(0deg) rotateY(0deg)";
      halo.style.opacity = 0.8;
      halo.style.transform = "scale(1)";
    });

    controls.pulse.addEventListener("click", () => {
      state.speed = state.speed >= 2.5 ? 1 : state.speed + 0.5;
      setSpeed(state.speed);
      panel.animate(
        [
          { transform: "scale(1)" },
          { transform: "scale(1.04)" },
          { transform: "scale(1)" },
        ],
        { duration: 600, easing: "ease-out" }
      );
    });

    controls.shuffle.addEventListener("click", () => {
      state.hue = (state.hue + 48 + Math.random() * 120) % 360;
      setHue(state.hue);
      const count = 10 + Math.floor(Math.random() * 8);
      spawnParticles(count);
      trace.style.animationDuration = `${9 + Math.random() * 6}s`;
    });

    // Initialize state
    setHue(state.hue);
    setSpeed(state.speed);
  </script>
